from dash import Dash, html, dcc
import pandas as pd
import plotly.express as px

# -----------------------
# DATA
# -----------------------

# Line chart data (Orders over time)
orders_df = pd.DataFrame({
    "month": ["Jan", "Feb", "Mar", "Apr"],
    "orders": [1200, 1500, 1800, 2100]
})

# Bar chart data (Products by category)
products_df = pd.DataFrame({
    "category": ["Electronics", "Furniture", "Fashion"],
    "count": [300, 180, 520]
})

# Pie chart data (Sentiment distribution)
sentiment_df = pd.DataFrame({
    "sentiment": ["Positive", "Neutral", "Negative"],
    "reviews": [800, 250, 150]
})

# -----------------------
# FIGURES
# -----------------------

fig_orders = px.line(
    orders_df,
    x="month",
    y="orders",
    title="Monthly Orders Trend"
)

fig_products = px.bar(
    products_df,
    x="category",
    y="count",
    title="Products by Category"
)

fig_sentiment = px.pie(
    sentiment_df,
    names="sentiment",
    values="reviews",
    title="Customer Review Sentiment"
)

# -----------------------
# DASH APP
# -----------------------

app = Dash(__name__)

app.layout = html.Div([
    html.H1("Olist E-Commerce Dashboard", style={'textAlign': 'center'}),

    dcc.Graph(figure=fig_orders),
    dcc.Graph(figure=fig_products),
    dcc.Graph(figure=fig_sentiment)
])

# -----------------------
# RUN APP
# -----------------------

if __name__ == "__main__":
    app.run(debug=True)
